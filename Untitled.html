    <!-- Load jsPDF library (for PDF generation) --><script src="https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js"></script>

    

    <script>

        // Define necessary global variables for the libraries to work inside the window scope

        const { jsPDF } = window.jspdf;


        /**

         * Generates the PDF document directly using jsPDF.

         */

        function generatePDF() {

            // 1. Get form values

            const name = document.getElementById('customerName').value;

            const email = document.getElementById('customerEmail').value;

            const phone = document.getElementById('customerPhone').value;

            

            // New Flooring Job Details

            const jobDate = document.getElementById('jobDate').value;

            const customerAddress = document.getElementById('customerAddress').value;

            const totalArea = document.getElementById('totalArea').value;

            const floorType = document.getElementById('floorType').value;

            const fittingAmount = document.getElementById('fittingAmount').value;


            const messageBox = document.getElementById('messageBox');

            messageBox.classList.add('hidden'); // Hide any previous messages


            // Simple validation for required fields

            if (!name || !email || !jobDate || !customerAddress || !totalArea || !floorType || !fittingAmount) {

                messageBox.textContent = 'Please fill out all required fields marked with (*).';

                messageBox.classList.remove('hidden', 'bg-green-100', 'text-green-700');

                messageBox.classList.add('bg-red-100', 'text-red-700');

                return;

            }


            // Format fitting amount as currency (using USD as a general example)

            const formattedAmount = new Intl.NumberFormat('en-US', {

                style: 'currency',

                currency: 'USD',

            }).format(fittingAmount);


            // Show loading state

            const generateButton = document.getElementById('generateButton');

            generateButton.innerHTML = `<svg class="animate-spin -ml-1 mr-3 h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">

                <circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle>

                <path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>

            </svg> Generating Report...`;

            generateButton.disabled = true;


            try {

                const pdf = new jsPDF({

                    orientation: 'portrait',

                    unit: 'mm',

                    format: 'a4'

                });


                let y_pos = 20; // Starting Y position


                // --- Header ---

                pdf.setFillColor(67, 56, 202); // Indigo-700 equivalent

                pdf.rect(0, 0, 210, 25, 'F'); // Full width header bar

                pdf.setFontSize(24);

                pdf.setTextColor(255, 255, 255); // White text

                pdf.text("Flooring Job Report", 105, 15, { align: 'center' }); // Centered title


                y_pos = 35; // Reset Y position after header


                pdf.setTextColor(55, 65, 81); // Gray-700

                pdf.setFontSize(10);

                pdf.text(`Report Generated: ${new Date().toLocaleDateString()} at ${new Date().toLocaleTimeString()}`, 10, y_pos);

                y_pos += 10;



                // --- Job & Customer Information ---

                pdf.setFillColor(243, 244, 246); // Gray-100 for section background

                pdf.rect(10, y_pos - 3, 190, 80, 'F'); // Background rectangle for the section

                pdf.setTextColor(30, 58, 138); // Darker blue for section title

                pdf.setFontSize(16);

                pdf.text("Job & Customer Information", 15, y_pos);

                y_pos += 10;

                

                pdf.setTextColor(0, 0, 0); // Black for main text

                pdf.setFontSize(12);

                pdf.text(`Job Date: ${jobDate}`, 15, y_pos);

                y_pos += 7;

                pdf.text(`Customer: ${name}`, 15, y_pos);

                y_pos += 7;

                pdf.text(`Email: ${email}`, 15, y_pos);

                y_pos += 7;

                pdf.text(`Phone: ${phone || 'N/A'}`, 15, y_pos);

                y_pos += 10;


                pdf.text("Address:", 15, y_pos);

                y_pos += 5;

                const addressLines = pdf.splitTextToSize(customerAddress, 80); // Wrap address if too long

                pdf.text(addressLines, 15, y_pos);

                y_pos += (addressLines.length * 6) + 10; // Adjust Y based on address lines



                // --- Installation Details ---

                pdf.setFillColor(243, 244, 246); // Gray-100 for section background

                pdf.rect(10, y_pos - 3, 190, 45, 'F'); // Background rectangle for the section

                pdf.setTextColor(30, 58, 138); // Darker blue for section title

                pdf.setFontSize(16);

                pdf.text("Installation Details", 15, y_pos);

                y_pos += 10;


                pdf.setTextColor(0, 0, 0); // Black for main text

                pdf.setFontSize(12);

                pdf.text(`Floor Type: ${floorType}`, 15, y_pos);

                y_pos += 7;

                pdf.text(`Total Area Fitted: ${totalArea} sq. ft./sq. m`, 15, y_pos);

                y_pos += 7;

                pdf.text(`Fitting Amount: ${formattedAmount}`, 15, y_pos);

                

                // Add color to fitting amount

                pdf.setTextColor(22, 163, 74); // Green-600

                pdf.text(formattedAmount, 15 + pdf.getTextWidth(`Fitting Amount: `), y_pos);

                pdf.setTextColor(0, 0, 0); // Reset color

                y_pos += 15;



                // --- Footer ---

                pdf.setFontSize(10);

                pdf.setTextColor(107, 114, 128); // Gray-500

                pdf.text("--- End of Flooring Job Report ---", 105, 280, { align: 'center' });



                // Download the PDF

                const filename = `Flooring_Report_${name.replace(/\s/g, '_')}_${jobDate}.pdf`;

                pdf.save(filename);


                // Show success message

                messageBox.textContent = `Successfully generated and downloaded PDF: ${filename}`;

                messageBox.classList.remove('hidden', 'bg-red-100', 'text-red-700');

                messageBox.classList.add('bg-green-100', 'text-green-700');

                

            } catch (error) {

                console.error("PDF Generation Error:", error);

                messageBox.textContent = 'An error occurred during PDF generation. Check console for details.';

                messageBox.classList.remove('hidden', 'bg-green-100', 'text-green-700');

                messageBox.classList.add('bg-red-100', 'text-red-700');

            } finally {

                // Reset button state

                generateButton.innerHTML = 'Generate PDF Report';

                generateButton.disabled = false;

            }

        }

    </script>

                <select id="floorType" required

            </div>

            

            <!-- Total Area Fitted --><div>

                <label for="totalArea" class="block text-sm font-medium text-gray-700 mb-1">Total Area Fitted (e.g., in sq. ft. or sq. m) <span class="text-red-500">*</span></label>

                <input type="number" id="totalArea" placeholder="e.g., 150.5" required min="0.01" step="0.01"

            <!-- Generate Button --><button type="button" onclick="generatePDF()" id="generateButton"
