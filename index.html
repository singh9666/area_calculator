            <button id="add-room-btn"

                class="w-full py-3 mt-6 bg-indigo-600 text-white font-semibold rounded-lg hover:bg-indigo-700 transition duration-150 shadow-md flex items-center justify-center">

        <section class="pt-8 border-t border-gray-200">

            <h2 class="text-2xl font-bold text-gray-800 mb-4">Results Summary</h2>


            <div class="bg-gray-50 p-6 rounded-lg shadow-inner">

                <div class="flex justify-between items-center mb-4">

                    <span class="text-xl font-medium text-gray-600">Total Calculated Area:</span>

                    <span id="total-area" class="text-3xl font-extrabold text-indigo-600">0.00 m²</span>

                </div>


                <h3 class="text-lg font-semibold text-gray-700 mt-6 mb-3">Individual Room Areas:</h3>

                <ul id="individual-areas" class="space-y-2 text-sm text-gray-600">

                    <li class="text-gray-400 italic">No rooms entered yet.</li>

                </ul>

            </div>

        </section>

    <script>

        const roomList = document.getElementById('room-list');

        const addRoomBtn = document.getElementById('add-room-btn');

        const totalAreaDisplay = document.getElementById('total-area');

        const individualAreasList = document.getElementById('individual-areas');

        let roomCount = 0;

        

        // Expose calculateTotalArea to the window scope for accessibility if needed, 

        // though we now rely on the event delegation listener below.

        window.calculateTotalArea = calculateTotalArea;


        /**

         * Adds a new room input block to the list.

         */

        function addRoom() {

            roomCount++;

            const id = `room-${roomCount}`;

            const roomDiv = document.createElement('div');

            roomDiv.className = 'flex flex-col sm:flex-row gap-4 items-center bg-gray-50 p-4 rounded-lg border border-gray-200';

            roomDiv.id = id;

            roomDiv.innerHTML = `

                <div class="flex-grow">

                    <label for="${id}-name" class="block text-xs font-medium text-gray-500">Room Name</label>

                    <input type="text" id="${id}-name" value="Room ${roomCount}"

                           class="w-full mt-1 p-2 border border-gray-300 rounded-md text-gray-800 input-field">

                </div>


                <div class="w-full sm:w-1/3">

                    <label for="${id}-length" class="block text-xs font-medium text-gray-500">Length (m)</label>

                    <input type="number" step="0.01" min="0" id="${id}-length" placeholder="Length"

                           class="w-full mt-1 p-2 border border-gray-300 rounded-md text-gray-800 input-field">

                </div>


                <div class="w-full sm:w-1/3">

                    <label for="${id}-width" class="block text-xs font-medium text-gray-500">Width (m)</label>

                    <input type="number" step="0.01" min="0" id="${id}-width" placeholder="Width"

                           class="w-full mt-1 p-2 border border-gray-300 rounded-md text-gray-800 input-field">

                </div>


                <button class="remove-room-btn flex-shrink-0 mt-4 sm:mt-0 p-2 text-red-500 hover:text-red-700 transition duration-150 rounded-full"

                        title="Remove Room" data-room-id="${id}" onclick="removeRoom('${id}')">

                    <svg xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24" stroke-width="2" stroke="currentColor" class="w-6 h-6">

                        <path stroke-linecap="round" stroke-linejoin="round" d="M6 18L18 6M6 6l12 12" />

                    </svg>

                </button>

            `;

            roomList.appendChild(roomDiv);

            document.getElementById(`${id}-length`).focus();

            calculateTotalArea(); // Calculate after adding

        }


        /**

         * Removes a specific room input block.

         * @param {string} id - The ID of the room block to remove.

         */

        window.removeRoom = function(id) {

            const roomDiv = document.getElementById(id);

            if (roomDiv) {

                roomDiv.remove();

                calculateTotalArea(); // Recalculate after removal

            }

        }


        /**

         * Calculates the total area and updates the results display.

         */

        function calculateTotalArea() {

            let totalArea = 0;

            const individualResults = [];

            const roomBlocks = roomList.querySelectorAll('[id^="room-"]');


            roomBlocks.forEach(roomBlock => {

                const id = roomBlock.id;

                const nameInput = document.getElementById(`${id}-name`);

                const lengthInput = document.getElementById(`${id}-length`);

                const widthInput = document.getElementById(`${id}-width`);


                const roomName = nameInput ? (nameInput.value || `Room ${id.split('-')[1]}`) : 'Unnamed Room';

                

                // Use a check to ensure we have input elements before trying to read values

                const length = lengthInput ? parseFloat(lengthInput.value) : NaN;

                const width = widthInput ? parseFloat(widthInput.value) : NaN;


                // Check for valid positive numbers

                if (!isNaN(length) && length > 0 && !isNaN(width) && width > 0) {

                    const area = length * width;

                    totalArea += area;


                    individualResults.push({

                        name: roomName,

                        length: length,

                        width: width,

                        area: area.toFixed(2)

                    });

                }

            });


            // Update Total Area Display

            totalAreaDisplay.textContent = `${totalArea.toFixed(2)} m²`;


            // Update Individual Areas List

            individualAreasList.innerHTML = '';

            if (individualResults.length === 0) {

                individualAreasList.innerHTML = '<li class="text-gray-400 italic">No rooms entered yet.</li>';

            } else {

                individualResults.forEach(result => {

                    const listItem = document.createElement('li');

                    listItem.className = 'flex justify-between border-b border-gray-100 py-1';

                    listItem.innerHTML = `

                        <span class="font-medium text-gray-700">${result.name} (${result.length}m x ${result.width}m)</span>

                        <span class="font-semibold text-indigo-500">${result.area} m²</span>

                    `;

                    individualAreasList.appendChild(listItem);

                });

            }

        }


        // Initialize the app with one room input and attach event listener

        document.addEventListener('DOMContentLoaded', () => {

            addRoom(); // Add first room on load

            addRoomBtn.addEventListener('click', addRoom);

            

            // FIX: Use event delegation on the room list container to capture all input changes (including dynamically added rooms).

            roomList.addEventListener('input', (event) => {

                // Trigger calculation if any relevant input field changes (name, length, or width)

                if (event.target.type === 'number' || event.target.id.endsWith('-name')) {

                    calculateTotalArea();

                }

            });

            

            calculateTotalArea(); 

        });


    </script>
